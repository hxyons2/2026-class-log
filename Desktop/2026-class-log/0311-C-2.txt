#include <stdio.h>   // 표준 입출력 함수 사용 (printf, scanf)

int main() {         // 프로그램 시작 함수
    
    int num;         // 사용자가 입력할 숫자를 저장할 정수 변수
    
    printf("숫자를 입력하세요:");  // 사용자에게 숫자 입력 안내
    scanf("%d", &num);            // 정수 입력 (%d), 변수의 주소(&num)에 저장
    
    if (num % 2 == 0) {           // num을 2로 나눈 나머지가 0이면 (짝수 조건)
        printf("짝수\n");          // 짝수라고 출력
        printf("2로 나눈 나머지가 0\n"); // 나머지가 0이라는 설명 출력
    }
    else {                        // 위 조건이 아니라면 (즉, 홀수라면)
        printf("홀수\n");          // 홀수라고 출력
        printf("2로 나눈 나머지가 1\n"); // 나머지가 1이라는 설명 출력
    }
    
    return 0;                     // 프로그램 정상 종료
}
