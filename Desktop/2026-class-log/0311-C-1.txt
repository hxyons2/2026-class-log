 #include <stdio.h>   // 표준 입출력 함수 사용 (printf, scanf)

int main() {         // 프로그램 시작 함수
    
    int d;           // 여행 일수 저장할 정수 변수
    float m;         // 여행 경비 저장할 실수 변수
    char gr;         // 여행 등급(A,B,C) 저장할 문자 변수
    char Tr[30];     // 여행지 이름 저장할 문자열 배열 (최대 29글자 + NULL)
    
    printf("다음 여행 정보를 입력받아 출력하시오."); // 안내 문장 출력
    
    printf("여행지 이름:\n");      // 여행지 이름 입력 안내
    scanf("%s", Tr);               // 문자열 입력 (%s 사용)
                                   // 문자열은 배열 이름 자체가 주소라 & 안 붙임
    
    printf("여행 일수:\n");        // 여행 일수 입력 안내
    scanf("%d", &d);               // 정수 입력 (%d), 변수 주소 필요
    
    printf("여행 경비:\n");        // 여행 경비 입력 안내
    scanf("%f", &m);               // 실수 입력 (%f), 변수 주소 필요
    
    printf("여행 등급(A, B, C):"); // 여행 등급 입력 안내
    scanf(" %c", &gr);             // 문자 입력 (%c)
                                   // %c 앞 공백: 이전 입력의 엔터 제거
    
    return 0;                      // 프로그램 정상 종료
}
