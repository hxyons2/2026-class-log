#include <stdio.h>   // 표준 입출력 함수(printf, scanf)를 사용하기 위한 헤더파일

int main()           // 프로그램 시작
{
    int s;           // 정수를 저장할 변수 s 선언
    
    printf("숫자를 입력하시오.: ");   // 사용자에게 숫자 입력 안내
    scanf("%d", &s);                // 입력한 정수를 변수 s에 저장
    
    while(1)        // 무한 반복 (1은 항상 참이므로 계속 반복됨)
    {
    
    if(s == 10){    // 입력한 숫자가 10과 같으면
        printf(" 정답입니다.\n");   // 정답 메시지 출력
        break;}     // while 반복문을 종료
        
    else if(s > 10) // 입력한 숫자가 10보다 크면
        printf(" 너무 높아요.\n");   // 너무 높다는 메시지 출력
    
    else            // 위 조건이 모두 아니면 (10보다 작으면)
        printf(" 너무 낮아요.");    // 너무 낮다는 메시지 출력
    }

    return 0;       // 프로그램 정상 종료
}
